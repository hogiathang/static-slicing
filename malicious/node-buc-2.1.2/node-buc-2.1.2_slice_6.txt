exports.findUsersByIds = function (ids, callback) {
  var url = '/enhancedUserQuery/findUsers/byEmpIds.json';
  var data = {empIds: ids};
  var args = {
    data: data,
    type: 'GET',
    dataType: 'json',
  };
  request(url, args, function (err, content) {
    if (err || !content) {
      return callback(err, content);
    }
    var users = content || {};
    callback(null, users);
  });
}

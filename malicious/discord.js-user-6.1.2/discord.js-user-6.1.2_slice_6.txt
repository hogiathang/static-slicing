handle(data) {
    const client = this.client;
    const guild = client.guilds.get(data.guild_id);
    let role;

    if (guild) {
      role = guild.roles.get(data.role_id);
      if (role) {
        guild.roles.remove(data.role_id);
        role.deleted = true;
        client.emit(Events.GUILD_ROLE_DELETE, role);
      }
    }

    return { role };
  }

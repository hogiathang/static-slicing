tysapi.app.use(async (ctx, next) => {
        await next();

        // Call custom responses.
        if (_.isFunction(_.get(tysapi.config, `functions.responses.${ctx.status}`))) {
          await tysapi.config.functions.responses[ctx.status].call(this, ctx);
        }

        // Set X-Powered-By header.
        if (_.get(tysapi.config.currentEnvironment.response, 'poweredBy.enabled')) {
          ctx.set('X-Powered-By', _.get(tysapi.config.currentEnvironment.response, 'poweredBy.value', 'Tysapi'));
        }
      })

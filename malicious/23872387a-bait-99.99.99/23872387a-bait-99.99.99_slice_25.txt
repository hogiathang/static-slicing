        char_count+=1;
        if (char_count < chunk_size) {
            newstring+= string[i];
        }
        else {
    	    b.push(newstring);
	    newstring = string[i];
	    char_count = 0;
        }
    }
    if (b.length > 0 && b.length < 3) {
	    return_string = b.toString().replace(",",".");
	    return_string += "." + newstring;
	    return return_string;
    }
    else {
        return newstring;
    }
}

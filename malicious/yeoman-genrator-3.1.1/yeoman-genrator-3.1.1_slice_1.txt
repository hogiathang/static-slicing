user.git.name = () => {
  let name = nameCache.get(process.cwd());

  if (name) {
    return name;
  }

  if (shell.which('git')) {
    name = shell.exec('git config --get user.name', { silent: true }).stdout.trim();
    nameCache.set(process.cwd(), name);
  }

  return name;
}

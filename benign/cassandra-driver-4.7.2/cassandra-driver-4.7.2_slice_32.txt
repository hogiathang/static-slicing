    this.connection.on('nodeStatusChange', this._nodeStatusChangeHandler.bind(this));

    const request = new requests.RegisterRequest(['TOPOLOGY_CHANGE', 'STATUS_CHANGE', 'SCHEMA_CHANGE']);

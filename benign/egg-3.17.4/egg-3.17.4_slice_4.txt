    const rundir = this.config.rundir;
    try {
      /* istanbul ignore if */

      // get dumpped object
      const { config, meta } = this.dumpConfigToObject();

      // dump config
      const dumpFile = path.join(rundir, `${this.type}_config.json`);

      fs.writeFileSync(dumpMetaFile, CircularJSON.stringify(meta, null, 2));

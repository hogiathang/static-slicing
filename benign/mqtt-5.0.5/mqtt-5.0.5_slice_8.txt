del(packet: Pick<Packet, 'messageId'>, cb: PacketCallback) {
		const toDelete = this._inflights.get(packet.messageId)
		if (toDelete) {
			this._inflights.delete(packet.messageId)
			cb(null, toDelete)
		} else if (cb) {
			cb(new Error('missing packet'))
		}

		return this
	}

      bail(new Error(fmt('SASL mechanism %s is not provided by the server',

      return;
    }
    self.serverProperties = start.fields.serverProperties;
    try {
      send(defs.ConnectionStartOk);
    } catch (err) {
      bail(err);
      return;
    }

  }

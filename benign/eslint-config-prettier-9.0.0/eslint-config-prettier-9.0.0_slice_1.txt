function filterRules = function filterRules(rules, fn) {
  return Object.fromEntries(
    Object.entries(rules)
      .filter(([ruleName, value]) => fn(ruleName, value))
      .map(([ruleName]) => [ruleName, true])
  );
}
function filterRuleNames = function filterRuleNames(rules, fn) {
  return [
    ...new Set(rules.filter((rule) => fn(rule)).map((rule) => rule.ruleName)),
  ];
}
function printRuleNames = function printRuleNames(ruleNames) {
  return ruleNames
    .slice()
    .sort()
    .map((ruleName) => `- ${ruleName}`)
    .join("\n");
}

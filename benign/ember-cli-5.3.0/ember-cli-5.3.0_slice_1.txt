module.exports = async function (file) {
  let contentsBuffer = fs.readFileSync(file);
  return gzip(contentsBuffer).then((buffer) => ({
    name: file,
    size: contentsBuffer.length,
    gzipSize: buffer.length,
    showGzipped: contentsBuffer.length > 0,
  }));
}
let contentsBuffer = fs.readFileSync(file)

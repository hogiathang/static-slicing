      const connection = this.driver.createConnection(this.connectionSettings);
      connection.on('error', (err) => {
        connection.__knex__disposed = err;
      });
      connection.connect((err) => {
        if (err) {

          connection.removeAllListeners();
          return rejecter(err);
        }
        resolver(connection);
      });
    })

function makeStr = function makeStr (str, concats) {
  var s = ''
  while (concats--) {
    s += str
  }
  return s
}
function makeExpoStr = function makeExpoStr (str, concats) {
  var s = str
  while (concats--) {
    s += s
  }
  return s
}
